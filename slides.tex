\section{Modell}
\labsectionframe

\begin{frame}{Linearisierte Bewegungsgleichung}
    Unter Vernachlaessigung der Corioliskraefte:
    \[
        \left(J_p + m_2 l_2^2\right)\ddot{\varphi}_2 + b_2\dot{\varphi}_2 + m_2 g l_2\sin\varphi_2
        = -m_2 L_1 l_2\cos\varphi_2\,\ddot{\varphi}_1
    \]
    Nahe des haengenden Arbeitspunktes gilt:\quad
    $\sin\varphi_2 \approx \varphi_2$,\; $\cos\varphi_2 \approx 1$.
\end{frame}
